     public void complete(){
         if(!isAchieved()){
             //can't complete achievements with the dev console shown.
             if(ui != null && ui.consolefrag != null && ui.consolefrag.shown() && !OS.username.equals("anuke") && this != openConsole) return;

         }
     }
